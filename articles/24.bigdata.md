size：每一行高度  remain: 6 条

```js
<Select v-model="selectData" style="width:200px">
  <virtual-list :size="30" :remain="6">
    <Option v-for="item in list" :value="item.value" :key="item.value">{{ item.label }}</Option>
</virtual-list>
</Select>
```
